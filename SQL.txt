CREATE TABLE karyawan (
nik INTEGER NOT NULL PRIMARY KEY,
namaDepan TEXT NOT NULL,
namaBelakang TEXT NOT NULL,
email TEXT NOT NULL,
phone TEXT,
kodeManajer INTEGER,
FOREIGN KEY(kodeManajer) REFERENCES manajer(kodeManajer));

CREATE TABLE departemen (
kodeDepartemen INTEGER NOT NULL PRIMARY KEY,
nama TEXT NOT NULL,
lokasi TEXT NOT NULL,
nik INTEGER NOT NULL,
FOREIGN KEY(nik) REFERENCES karyawan(nik));

CREATE TABLE manajer (
kodeManajer INTEGER NOT NULL PRIMARY KEY,
lamaJabatan TEXT,
kodeDepartemen INTEGER NOT NULL,
FOREIGN KEY(kodeDepartemen) REFERENCES departemen(kodeDepartemen));

CREATE TABLE proyek (
idProyek INTEGER NOT NULL PRIMARY KEY,
namaProyek TEXT NOT NULL,
waktuProyek TEXT NOT NULL,
nilaiProyek INTEGER NOT NULL);

CREATE TABLE proyekKaryawan (
idProyekNik INTEGER NOT NULL PRIMARY KEY,
nik INTEGER NOT NULL,
idProyek INTEGER NOT NULL,
FOREIGN KEY(nik) REFERENCES karyawan(nik),
FOREIGN KEY(idProyek) REFERENCES proyek(idProyek));
