CREATE TABLE Employee (id integer NOT NULL PRIMARY KEY AUTOINCREMENT, nik varchar, first_name varchar, last_name varchar, gender varchar, email varchar, phone_number varchar, departmentId integer, FOREIGN KEY (departmentId) REFERENCES Department(id));


CREATE TABLE Department (id integer NOT NULL PRIMARY KEY AUTOINCREMENT, name varchar, managerId integer, FOREIGN KEY (managerId) REFERENCES Manager(id));

CREATE TABLE Manager (id integer NOT NULL PRIMARY KEY AUTOINCREMENT, employeeId integer,FOREIGN KEY (employeeId) REFERENCES Employee(id));

CREATE TABLE Project (id integer NOT NULL PRIMARY KEY AUTOINCREMENT, name varchar);

CREATE TABLE ProjectEmployee (id integer NOT NULL PRIMARY KEY AUTOINCREMENT, employeeId integer, projectId integer, FOREIGN KEY (employeeId) REFERENCES Employee(id), FOREIGN KEY (projectId) REFERENCES Project(id));
